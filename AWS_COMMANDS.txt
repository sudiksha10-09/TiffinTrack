# Copy and paste these commands on your AWS server

# 1. Go to project directory
cd ~/TiffinTrack

# 2. Pull latest code
git pull origin main

# 3. Install python-dotenv if not already installed
pip install python-dotenv

# 4. Run the database fix
python3 fix_db_description.py

# 5. Restart the application
sudo systemctl restart tiffintrack

# 6. Check if it's running
sudo systemctl status tiffintrack

# 7. Check logs for errors
sudo journalctl -u tiffintrack -n 20 --no-pager

# Done! Now try adding your Gulab Jamun plan in the admin panel.
